b = Buffer.loadDialog(s); //you need to change the path to a soundfile in your computer

(
~tempo = 95; //change to bpm of track
~signature = 4;
~barlength= ~tempo/120;
~spb= (~barlength/~signature);
~bps= (~spb).reciprocal;

~length= b.numFrames/44100.0;

~bars= (~length/~barlength).round.asInteger;

~framesperbar= ~barlength*44100.0;

SynthDef(\stepplaybuf2,{arg bufnum=0,outbus=0,rate=1,startPos=0,dur=0.1,atk=0.005, rel=0.05, curve=0;
var playbuf, env;
playbuf= PlayBuf.ar(2,bufnum,BufRateScale.kr(bufnum)*rate,1,startPos,1);
env= EnvGen.ar(Env.linen(atk,(dur-atk-rel),rel, 1.0, curve),doneAction:2);
Out.ar(outbus,playbuf*env);
}).send(s);

SynthDef(\step15bass,{arg outbus=0,dur=0.2, index=2, modfreq=15;
var playbuf;
playbuf= LFTri.ar(SinOsc.ar(modfreq,0,modfreq*Line.kr(index,0,dur),(Line.kr(44+12,32,dur)).midicps),0, EnvGen.ar(Env([0,1,1,0],[0.001,dur*0.5,dur*0.5]),doneAction:2));

Out.ar(outbus,Pan2.ar(playbuf,0));
}).send(s);

SynthDef(\step15fx1,{arg panrate1=15, panrate2=0.15, delay=0.01, decay=1.5876;
var input;
input= In.ar(0,2);
input= Pan2.ar(input[0],SinOsc.ar(panrate1,0))+ Pan2.ar(input[1], SinOsc.ar(panrate2,0));
Out.ar(0,EnvGen.ar(Env([0,0.5,0.5,0],[~barlength, 2*~barlength, ~barlength]),doneAction:2)*CombN.ar(input,~spb,delay, decay));
}).send(s);

SynthDef(\step15fx2,{arg amp=0.5;
var input;
input= In.ar(0,2);
Out.ar(0,EnvGen.ar(Env([0,1,1,0]*amp,[0.001,~barlength,0.01]),doneAction:2)*Limiter.ar(LPF.ar(Rand(5,10)*input,500)));
}).send(s);

SynthDef(\step15fx3,{arg amp=0.1, wsize=0.05, ratio=1, pdisp=0.0, tdisp=0;
var input;
input= In.ar(0,2);
ReplaceOut.ar(0,EnvGen.ar(Env([0,1,1,0]*amp,[0.001,~barlength,0.01]),doneAction:2)*PitchShift.ar(input, wsize, ratio, pdisp, tdisp));
}).send(s);

SynthDef(\step15limiter,{
var input;
input=In.ar(0,2);
ReplaceOut.ar(0,Limiter.ar(FreeVerb.ar(input,0.2), 0.99,0.01));
}).send(s);
)

});
